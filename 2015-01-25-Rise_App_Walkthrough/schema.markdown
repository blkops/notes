## Event

  * contactName: String,
  * contactEmail: String,
  * locationName: String,
  * eventName: String,
  * url: String,
  * description: String,
  * address: String,
  * city: String,
  * state: String,
  * postcode: String,
  * latitude: Number,
  * longitude: Number,
  * country: String,
  * date: Date,
  * time: Date,
  * hashtag: [String]
